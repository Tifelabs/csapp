     \textbf{Perform the following number conversions}
     \vspace{0.8em}
     \begin{enumerate}[label=(\Alph*)]
        \item 0x25B9D2 to Binary
        \item binary 1010111001001001 to Hexadecimal
        \item 0xA8B3D to Binary
        \item binary 1100100010110110010110 to Hexadecimal
     \end{enumerate}
     
     \section*{\small Solution A}
     \vspace{0.2em}
     \small
     \begin{tabular}{l|cccccc}
        \toprule
        \textbf{Hexadecimal} & 2 & 5 & B & 9 & D & 2 \\
        \textbf{Binary}      &0010 & 0101 & 1101 & 1001 & 1101 & 0010\\
        \bottomrule
     \end{tabular}

     \section*{\small Solution B}
     \vspace{0.2em}
     \begin{tabular}{l|cccc}
        \toprule
        \textbf{Binary} & 1010 & 1110 & 0100 & 1001 \\
        \textbf{Hexadecimal} & A & E & 4 & 9 \\
        \bottomrule
     \end{tabular}

     \section*{\small Solution C}
     \vspace{0.2em}
     \begin{tabular}{l|ccccc}
        \toprule
        \textbf{Hexadecimal} & A & 8 & B & 3 & D \\
        \textbf{Binary} & 1010 & 1000 & 1011 & 0011 & 1101\\
        \bottomrule
     \end{tabular}

     \vspace{0.2em}

            \textbf{Binary} & 11 & 0010 & 0010 & 1101 & 1001 & 0110 \\
            \textbf{Hexadecimal} & 3 & 2 & 2 & D & 9 & 6 \\
            \bottomrule
        \end{tabular}
