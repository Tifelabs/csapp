\begin{flushleft}
\textbf{\small PRINCIPLE:} \texttt{Unsigned Multiplication}\\
For  $x$ and $y$ such that $0 \leq x, y \leq Umax_w$:
    \[
        x \ast ^u_wy = (x \cdot y) mod 2^w
    \]

\smallskip
\textbf{\small PRINCIPLE:} \texttt{Two's complement Multiplication}\\
For  $x$ and $y$ such that $Tmin_w \leq x, y \leq Tmax_w$:
    \[
        x \ast ^u_wy = U2T_w((x \cdot y) mod 2^w)
    \]

\textbf{\small DERIAVATION:} \texttt{Unsigned negation}\\
    When $x = 0$, the additive inverse is clearly $0$. For $x > 0$, consider
    the value $2^w-x$.Obseerver that this number is in the range $0 < 2^w - x < 2^w$. We can
    see that $(x+2^w-x) mod 2^w = 2^w mod 2^w = 0$
\end{flushleft}
